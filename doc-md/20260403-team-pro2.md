// ===== NormalizeTeamID — 团队 ID 归一化 =====
func NormalizeTeamID(value string) string {
    value = strings.ToLower(strings.TrimSpace(value))
    value = strings.ReplaceAll(value, "/", "-")
    value = strings.ReplaceAll(value, "_", "-")
    value = strings.ReplaceAll(value, " ", "-")
    for strings.Contains(value, "--") {
        value = strings.ReplaceAll(value, "--", "-")
    }
    return strings.Trim(value, "-")
}

// ===== normalizeChannelID — 频道 ID 归一化 =====
func normalizeChannelID(value string) string {
    value = NormalizeTeamID(value)
    if value == "" {
        return "main"
    }
    return value
}

// ===== OpenStore — 创建 Store 实例 =====
func OpenStore(storeRoot string) (*Store, error) {
    root := filepath.Join(strings.TrimSpace(storeRoot), "team")
    if err := os.MkdirAll(root, 0o755); err != nil {
        return nil, err
    }
    return &Store{root: root}, nil
}

// ===== withTeamLock — 文件级排他锁 =====
func (s *Store) withTeamLock(teamID string, fn func() error) error {
    if s == nil {
        return errors.New("nil team store")
    }
    teamID = NormalizeTeamID(teamID)
    if teamID == "" {
        return errors.New("empty team id")
    }
    if err := os.MkdirAll(filepath.Join(s.root, teamID), 0o755); err != nil {
        return err
    }
    lockFile, err := os.OpenFile(s.teamLockPath(teamID), os.O_CREATE|os.O_RDWR, 0o644)
    if err != nil {
        return err
    }
    defer lockFile.Close()
    if err := syscall.Flock(int(lockFile.Fd()), syscall.LOCK_EX); err != nil {
        return err
    }
    defer func() { _ = syscall.Flock(int(lockFile.Fd()), syscall.LOCK_UN) }()
    return fn()
}

// ===== readLastJSONLLines — 从文件尾部反向读取 N 行 =====
func readLastJSONLLines(path string, limit int) ([]string, error) {
    if limit <= 0 {
        return nil, nil
    }
    file, err := os.Open(path)
    if err != nil {
        return nil, err
    }
    defer file.Close()
    info, err := file.Stat()
    if err != nil {
        return nil, err
    }
    if info.Size() == 0 {
        return nil, nil
    }
    lines := make([]string, 0, limit)
    current := make([]byte, 0, 256)
    for pos := info.Size() - 1; pos >= 0 && len(lines) < limit; pos-- {
        var b [1]byte
        if _, err := file.ReadAt(b[:], pos); err != nil {
            return nil, err
        }
        if b[0] == '\n' {
            if line := strings.TrimSpace(reverseBytesToString(current)); line != "" {
                lines = append(lines, line)
            }
            current = current[:0]
            continue
        }
        current = append(current, b[0])
        if pos == 0 {
            if line := strings.TrimSpace(reverseBytesToString(current)); line != "" {
                lines = append(lines, line)
            }
        }
    }
    return lines, nil
}

// ===== channelPath — 频道 JSONL 文件路径 =====
func (s *Store) channelPath(teamID, channelID string) string {
    return filepath.Join(s.root, teamID, "channels", normalizeChannelID(channelID)+".jsonl")
}

  func (s *Store) loadMessagesFromShards(teamID, channelID string, limit int) ([]Message, error) {
      dir := s.channelShardDir(teamID, channelID)
      entries, err := os.ReadDir(dir)
      if err != nil {
          return nil, err
      }
      // 按文件名倒序（最新周在前）
      sort.Slice(entries, func(i, j int) bool {
          return entries[i].Name() > entries[j].Name()
      })

      var out []Message
      for _, entry := range entries {
          if !strings.HasSuffix(entry.Name(), ".jsonl") || entry.IsDir() {
              continue
          }
          path := filepath.Join(dir, entry.Name())
          remaining := limit - len(out)
          if limit <= 0 {
              remaining = 0 // 0 表示全部读取
          }
          lines, err := readLastJSONLLines(path, remaining)
          if err != nil {
              continue // 跳过损坏的分片
          }
          for _, line := range lines {
              var msg Message
              if json.Unmarshal([]byte(line), &msg) == nil {
                  out = append(out, msg)
              }
          }
          if limit > 0 && len(out) >= limit {
              break
          }
      }
      // 按时间倒序
      sort.SliceStable(out, func(i, j int) bool {
          return out[i].CreatedAt.After(out[j].CreatedAt)
      })
      if limit > 0 && len(out) > limit {
          out = out[:limit]
      }
      return out, nil
  }

步骤 7: 实现迁移工具（将旧单文件拆为分片）:

  func (s *Store) MigrateChannelToShards(teamID, channelID string) error {
      return s.withTeamLock(teamID, func() error {
          teamID = NormalizeTeamID(teamID)
          channelID = normalizeChannelID(channelID)
          legacyPath := s.channelPath(teamID, channelID)

          // 检查旧文件是否存在
          if _, err := os.Stat(legacyPath); os.IsNotExist(err) {
              return nil // 无需迁移
          }

          // 读取全部消息
          file, err := os.Open(legacyPath)
          if err != nil {
              return err
          }
          defer file.Close()

          scanner := bufio.NewScanner(file)
          scanner.Buffer(make([]byte, 0, 256*1024), 1024*1024)

          shardFiles := make(map[string]*os.File) // shardPath → file
          defer func() {
              for _, f := range shardFiles {
                  f.Close()
              }
          }()

          for scanner.Scan() {
              line := strings.TrimSpace(scanner.Text())
              if line == "" {
                  continue
              }
              var msg Message
              if err := json.Unmarshal([]byte(line), &msg); err != nil {
                  continue // 跳过损坏行
              }
              shardPath := s.channelShardPath(teamID, channelID, msg.CreatedAt)
              sf, ok := shardFiles[shardPath]
              if !ok {
                  if err := os.MkdirAll(filepath.Dir(shardPath), 0o755); err != nil {
                      return err
                  }
                  sf, err = os.OpenFile(shardPath, os.O_CREATE|os.O_APPEND|os.O_WRONLY, 0o644)
                  if err != nil {
                      return err
                  }
                  shardFiles[shardPath] = sf
              }
              sf.Write(append([]byte(line), '\n'))
          }

          // 重命名旧文件为 .bak
          return os.Rename(legacyPath, legacyPath+".bak")
      })
  }

  // verifyMessageSignature 验证消息的 Ed25519 签名
  // 签名覆盖: AuthorAgentID|ChannelID|Content|CreatedAt(RFC3339)
  func verifyMessageSignature(msg Message) bool {
      if msg.OriginPublicKey == "" || msg.Signature == "" {
          return false
      }
      pubKeyBytes, err := hex.DecodeString(msg.OriginPublicKey)
      if err != nil || len(pubKeyBytes) != ed25519.PublicKeySize {
          return false
      }
      sigBytes, err := hex.DecodeString(msg.Signature)
      if err != nil || len(sigBytes) != ed25519.SignatureSize {
          return false
      }
      payload := fmt.Sprintf("%s|%s|%s|%s",
          msg.AuthorAgentID,
          msg.ChannelID,
          msg.Content,
          msg.CreatedAt.UTC().Format(time.RFC3339))
      return ed25519.Verify(pubKeyBytes, []byte(payload), sigBytes)
  }

  package team

  // Task 状态常量（对齐 A2A TaskState 语义）
  const (
      TaskStateOpen      = "open"       // 已提交，等待处理
      TaskStateDoing     = "doing"      // 处理中
      TaskStateBlocked   = "blocked"    // 需要输入/等待
      TaskStateDone      = "done"       // 完成（终态）
      TaskStateFailed    = "failed"     // 失败（终态）
      TaskStateCancelled = "cancelled"  // 取消（终态）
      TaskStateRejected  = "rejected"   // 拒绝（终态）
  )

  // defaultTaskTransitions 默认状态转换表
  var defaultTaskTransitions = map[string][]string{
      TaskStateOpen:      {TaskStateDoing, TaskStateCancelled, TaskStateRejected},
      TaskStateDoing:     {TaskStateDone, TaskStateFailed, TaskStateBlocked, TaskStateCancelled},
      TaskStateBlocked:   {TaskStateDoing, TaskStateCancelled},
      TaskStateDone:      {TaskStateOpen},       // 允许 reopen
      TaskStateFailed:    {TaskStateOpen},        // 允许 retry
      TaskStateCancelled: {TaskStateOpen},        // 允许 reopen
      TaskStateRejected:  {},                     // 终态，不可变
  }

  // IsTerminalState 判断是否为终态
  func IsTerminalState(state string) bool {
      switch state {
      case TaskStateDone, TaskStateFailed, TaskStateCancelled, TaskStateRejected:
          return true
      }
      return false
  }

  // IsValidTransition 检查状态转换是否合法
  func IsValidTransition(from, to string) bool {
      allowed, ok := defaultTaskTransitions[from]
      if !ok {
          return false
      }
      for _, s := range allowed {
          if s == to {
              return true
          }
      }
      return false
  }

  // IsValidTransitionWithPolicy 支持 Policy 自定义转换规则
  func IsValidTransitionWithPolicy(policy Policy, from, to string) bool {
      if len(policy.TaskTransitions) > 0 {
          rule, ok := policy.TaskTransitions[from]
          if ok {
              for _, s := range rule.AllowedNext {
                  if s == to {
                      return true
                  }
              }
              return false
          }
      }
      // 回退到默认规则
      return IsValidTransition(from, to)
  }

  func (s *Store) Subscribe(teamID string) chan TeamEvent {
      s.subMu.Lock()
      defer s.subMu.Unlock()
      ch := make(chan TeamEvent, 32)
      s.subscribers[teamID] = append(s.subscribers[teamID], ch)
      return ch
  }

  func (s *Store) Unsubscribe(teamID string, ch chan TeamEvent) {
      s.subMu.Lock()
      defer s.subMu.Unlock()
      subs := s.subscribers[teamID]
      for i, sub := range subs {
          if sub == ch {
              s.subscribers[teamID] = append(subs[:i], subs[i+1:]...)
              close(ch)
              return
          }
      }
  }

      for _, ch := range s.subscribers[event.TeamID] {
          select {
          case ch <- event:
          default: // 丢弃慢消费者，不阻塞
          }
      }

  func (p Policy) Allows(action string, role string) bool {
      if len(p.Permissions) == 0 {
          // 无 Permissions 配置时，回退到旧的角色列表
          return p.legacyAllows(action, role)
      }
      roles, ok := p.Permissions[action]
      if !ok {
          return p.legacyAllows(action, role)
      }
      for _, r := range roles {
          if r == role || r == "*" {
              return true
          }
      }
      return false
  }

  // legacyAllows 兼容旧的粗粒度角色列表
  func (p Policy) legacyAllows(action string, role string) bool {
      var allowed []string
      switch {
      case strings.HasPrefix(action, "message."):
          allowed = p.MessageRoles
      case strings.HasPrefix(action, "task."):
          allowed = p.TaskRoles
      default:
          // 默认只允许 owner
          allowed = []string{"owner"}
      }
      if len(allowed) == 0 {
          return true // 未配置时默认允许
      }
      for _, r := range allowed {
          if r == role || r == "*" {
              return true
          }
      }
      return false
  }

      configs := s.loadWebhookConfigs(event.TeamID)
      for _, cfg := range configs {
          if matchesEventFilter(cfg.Events, event.Type) {
              go s.sendWebhook(cfg, event) // 异步，不阻塞
          }
      }
  }

  func (s *Store) sendWebhook(cfg PushNotificationConfig, event TeamEvent) {
      body, _ := json.Marshal(event)
      req, err := http.NewRequest("POST", cfg.URL, bytes.NewReader(body))
      if err != nil {
          return
      }
      req.Header.Set("Content-Type", "application/json")
      if cfg.Token != "" {
          req.Header.Set("Authorization", "Bearer "+cfg.Token)
      }
      client := &http.Client{Timeout: 10 * time.Second}
      resp, err := client.Do(req)
      if err != nil {
          return
      }
      resp.Body.Close()
  }

## 七、设计原则

1. **文件即数据库**：JSONL/JSON 是权威存储，Redis 仅做缓存
2. **签名即信任**：写操作可选签名验证，历史本身是可验证审计链
3. **追加不可变**：消息和历史只追加不修改，任务通过追加新版本更新
4. **渐进兼容**：所有新字段 `omitempty`，旧数据无需迁移即可读取
5. **模块边界**：Team 不依赖 Live/Topics，跨模块只走 Reference 引用
6. **A2A 对齐但不照搬**：借鉴状态机、AgentCard、Part、Context 概念，保持 P2P + 文件优先的本地特色
