## 一、Team 总览页和常用操作收口

### 问题分析

当前 `team.html`（142 行）和 `team_detail.html`（618 行）存在以下问题：

1. **首页(team.html)信息密度低**：只有团队数、成员总数、最近更新三个统计，没有展示用户最关心的"我现在该做什么"

// CountTasksByStatusCtx 返回各状态的任务计数
// 实现: 只读 tasks.index.json (不读 tasks.data.jsonl)
func (s *Store) CountTasksByStatusCtx(ctx context.Context, teamID string) (map[string]int, error) {

**1. 状态分级逻辑（handler_sync.go）**

```go
type teamSyncHealthLevel string
const (
    SyncHealthy  teamSyncHealthLevel = "healthy"   // 无冲突、无 dead-letter、最近有 publish/apply
    SyncAttention teamSyncHealthLevel = "attention" // 有未处理冲突或 dead-letter
    SyncStale    teamSyncHealthLevel = "stale"      // 超过 1h 无 publish 也无 apply
    SyncDisabled teamSyncHealthLevel = "disabled"   // 无订阅 Team
)

func determineTeamSyncHealth(status corehaonews.SyncTeamSyncStatus, webhook teamcore.WebhookDeliveryStatus) teamSyncHealthLevel {
    if status.SubscribedTeams == 0 { return SyncDisabled }
    if status.Conflicts > status.ResolvedConflicts || webhook.DeadLetterCount > 0 {
        return SyncAttention
    }
    if status.LastPublishedAt != nil {
        if time.Since(*status.LastPublishedAt) > time.Hour { return SyncStale }
    }
    return SyncHealthy
}
```

func (s *Store) LoadRecentDeliveriesCtx(ctx context.Context, teamID string, limit int) ([]WebhookDeliveryRecord, error) {
    // 已有 webhook_delivery.go 中的结构，只需暴露读取方法
    records, err := s.loadWebhookDeliveries(teamID)
    if err != nil { return nil, err }
    // 按时间倒序，取 limit 条
    sort.Slice(records, func(i, j int) bool {
        return records[i].CreatedAt.After(records[j].CreatedAt)
    })
    if len(records) > limit { records = records[:limit] }
    return records, nil
}

func writeTeamError(w http.ResponseWriter, status int, resp TeamErrorResponse) {

(function() {
  const teamID = '{{.Team.TeamID}}';
  const indicator = document.getElementById('sse-indicator');
  const es = new EventSource('/api/teams/' + teamID + '/events');

  es.onopen = function() {

      switch(event.kind) {
        case 'message': text = '新消息'; break;
        case 'task':    text = '任务更新: ' + event.action; break;
        case 'member':  text = '成员变更'; break;
        case 'artifact':text = '新产物'; break;
        default:        text = event.kind + '.' + event.action;
      }
      toast.textContent = text;
      toast.classList.add('show');
      setTimeout(() => toast.classList.remove('show'), 3000);
    } catch(err) {}
  };

  // 页面离开时关闭

})();
</script>

### 问题分析

### 方案 P9-1：Team 内统一搜索 API + 页面

**目标**：一个搜索框，同时搜消息/任务/产物/历史

**设计约束**：
- 明文 JSONL 存储，不引入外部搜索引擎（不加 Elasticsearch、不加 SQLite FTS）
- 做法：遍历文件 + `strings.Contains` / `strings.EqualFold` 匹配，带合理的 limit 和范围限制
- 对大 Team（数万条消息）可能较慢，但这是 P2P 明文系统的取舍——保持简单

**Store 层新增**（新文件 `search.go`）：

```go
package team

import (
    "context"
    "strings"
    "time"
)

type SearchResult struct {
    Kind      string    `json:"kind"`       // "message" / "task" / "artifact" / "history"
    ID        string    `json:"id"`
    Title     string    `json:"title"`      // 任务/产物标题，消息取前 80 字符
    Snippet   string    `json:"snippet"`    // 匹配内容的前后文截取
    ChannelID string    `json:"channel_id,omitempty"`
    Author    string    `json:"author,omitempty"`
    CreatedAt time.Time `json:"created_at"`
}

type SearchQuery struct {
    Keyword   string    // 必填
    Kind      string    // 可选：限定只搜某类（message/task/artifact/history）
    ChannelID string    // 可选：限定频道
    Author    string    // 可选：限定作者
    Since     time.Time // 可选：时间范围开始
    Until     time.Time // 可选：时间范围结束
    Limit     int       // 默认 30
}

func (s *Store) SearchCtx(ctx context.Context, teamID string, q SearchQuery) ([]SearchResult, error) {
    if q.Limit <= 0 { q.Limit = 30 }
    if q.Limit > 100 { q.Limit = 100 }
    keyword := strings.ToLower(q.Keyword)
    var results []SearchResult

    // 1. 搜消息（只搜最近 N 条或指定频道/时间范围）
    if q.Kind == "" || q.Kind == "message" {
        channels, _ := s.ListChannelsCtx(ctx, teamID)
        for _, ch := range channels {
            if q.ChannelID != "" && ch.ChannelID != q.ChannelID { continue }
            // 加载最近 500 条消息（或用 Since/Until 限制范围）
            msgs, _ := s.LoadMessagesCtx(ctx, teamID, ch.ChannelID, 500)
            for _, msg := range msgs {
                if ctx.Err() != nil { return results, ctx.Err() }
                if !q.Since.IsZero() && msg.CreatedAt.Before(q.Since) { continue }
                if !q.Until.IsZero() && msg.CreatedAt.After(q.Until) { continue }
                if q.Author != "" && msg.AuthorAgentID != q.Author { continue }
                if strings.Contains(strings.ToLower(msg.Content), keyword) {
                    results = append(results, SearchResult{
                        Kind:      "message",
                        ID:        msg.MessageID,
                        Title:     truncate(msg.Content, 80),
                        Snippet:   extractSnippet(msg.Content, keyword, 120),
                        ChannelID: msg.ChannelID,
                        Author:    msg.AuthorAgentID,
                        CreatedAt: msg.CreatedAt,
                    })
                    if len(results) >= q.Limit { return results, nil }
                }
            }
        }
    }

    // 2. 搜任务
    if q.Kind == "" || q.Kind == "task" {
        tasks, _ := s.LoadTasksCtx(ctx, teamID, 0) // 全量加载
        for _, task := range tasks {
            if strings.Contains(strings.ToLower(task.Title), keyword) ||
               strings.Contains(strings.ToLower(task.Description), keyword) {
                results = append(results, SearchResult{
                    Kind:      "task",
                    ID:        task.TaskID,
                    Title:     task.Title,
                    Snippet:   extractSnippet(task.Description, keyword, 120),
                    ChannelID: task.ChannelID,
                    Author:    task.CreatedBy,
                    CreatedAt: task.CreatedAt,
                })
                if len(results) >= q.Limit { return results, nil }
            }
        }
    }

    // 3. 搜产物
    if q.Kind == "" || q.Kind == "artifact" {
        artifacts, _ := s.LoadArtifactsCtx(ctx, teamID, 0)
        for _, art := range artifacts {
            if strings.Contains(strings.ToLower(art.Title), keyword) ||
               strings.Contains(strings.ToLower(art.Content), keyword) ||
               strings.Contains(strings.ToLower(art.Summary), keyword) {
                results = append(results, SearchResult{
                    Kind:      "artifact",
                    ID:        art.ArtifactID,
                    Title:     art.Title,
                    Snippet:   extractSnippet(art.Content, keyword, 120),
                    ChannelID: art.ChannelID,
                    Author:    art.CreatedBy,
                    CreatedAt: art.CreatedAt,
                })
                if len(results) >= q.Limit { return results, nil }
            }
        }
    }

    // 4. 搜历史
    if q.Kind == "" || q.Kind == "history" {
        history, _ := s.LoadHistoryCtx(ctx, teamID, 500)
        for _, event := range history {
            if strings.Contains(strings.ToLower(event.Summary), keyword) {
                results = append(results, SearchResult{
                    Kind:      "history",
                    ID:        event.EventID,
                    Title:     event.Scope + "." + event.Action,
                    Snippet:   event.Summary,
                    Author:    event.ActorAgentID,
                    CreatedAt: event.CreatedAt,
                })
                if len(results) >= q.Limit { return results, nil }
            }
        }
    }

    return results, nil
}

// extractSnippet 在 content 中找到 keyword 并返回前后各 N 个字符的上下文
func extractSnippet(content, keyword string, maxLen int) string {
    lower := strings.ToLower(content)
    idx := strings.Index(lower, keyword)
    if idx < 0 { return truncate(content, maxLen) }
    start := idx - 40
    if start < 0 { start = 0 }
    end := idx + len(keyword) + 40
    if end > len(content) { end = len(content) }
    snippet := content[start:end]
    if start > 0 { snippet = "..." + snippet }
    if end < len(content) { snippet = snippet + "..." }
    return snippet
}

func truncate(s string, maxLen int) string {
    runes := []rune(s)
    if len(runes) <= maxLen { return s }
    return string(runes[:maxLen]) + "..."
}
```

        if t, err := time.Parse(time.RFC3339, v); err == nil { q.Since = t }
    }
    results, err := store.SearchCtx(r.Context(), teamID, q)

        return
    }
    newsplugin.WriteJSON(w, http.StatusOK, map[string]any{
        "scope":   "team-search",
        "team_id": teamID,
        "query":   q.Keyword,
        "count":   len(results),

// 在加载 history 后，计算本周摘要
weekStart := beginOfWeek(time.Now())
var weekSummary teamHistoryWeekSummary
for _, event := range history {
    if event.CreatedAt.Before(weekStart) { break } // history 已按时间倒序
    weekSummary.Total++
    switch event.Scope {
    case "message": weekSummary.Messages++
    case "task":    weekSummary.Tasks++
    case "member":  weekSummary.Members++
    case "artifact":weekSummary.Artifacts++
    case "policy":  weekSummary.Policies++
    case "channel": weekSummary.Channels++
    }
    // 标记"关键变更"（Policy 变更、成员角色变更、任务完成）
    if event.Scope == "policy" ||
       (event.Scope == "member" && event.Action == "update") ||
       (event.Scope == "task" && event.Action == "update" && isImportantTaskChange(event)) {
        weekSummary.KeyChanges = append(weekSummary.KeyChanges, event)
    }
}
data.WeekSummary = weekSummary

func (s *Store) CreateTeamFromTemplate(ctx context.Context, teamID string, templateID string, ownerAgentID string) error {
    tpl, err := s.loadTemplate(templateID)
    if err != nil { return err }

    // 1. 创建 Team info
    info := Info{
        TeamID: teamID, Title: tpl.Name, Description: tpl.Description,
        OwnerAgentID: ownerAgentID, Visibility: "private",
        Channels: channelIDs(tpl.Channels),
        CreatedAt: time.Now(), UpdatedAt: time.Now(),
    }
    if err := s.SaveTeamCtx(ctx, teamID, info); err != nil { return err }

    // 2. 创建频道
    for _, ch := range tpl.Channels {
        s.SaveChannelCtx(ctx, teamID, ch)
    }

    // 3. 设置 Policy
    if tpl.Policy != nil {
        s.SavePolicyCtx(ctx, teamID, *tpl.Policy)
    }

    // 4. 创建初始任务
    for _, task := range tpl.InitialTasks {
        task.TeamID = teamID
        task.TaskID = generateID()
        task.CreatedBy = ownerAgentID
        task.CreatedAt = time.Now()
        s.SaveTaskCtx(ctx, teamID, task)
    }

    // 5. 添加 owner 为成员
    s.SaveMembersCtx(ctx, teamID, []Member{{
        AgentID: ownerAgentID, Role: "owner", Status: "active", JoinedAt: time.Now(),
    }})

    return nil
}

## 原则重申

1. **不加新的底层能力**：上述所有方案都是在已有实底座上做 UI/UX/提示/视图 层的改善
2. **不破坏明文 + P2P**：所有数据仍然是 JSONL 明文、所有同步仍然是 P2P 直连
3. **不做空壳**：每个方案都明确了要改哪个文件哪几行、写什么代码、怎么验证
4. **增量可测**：每批完成后都有具体的手动验证步骤，不依赖额外基础设施
5. **保持三模块隔离**：Team / Live / Topics 不发生任何耦合
6. **新增字段向后兼容**：DueAt、ParentTaskID 都是 omitempty 可选字段，不影响现有数据
